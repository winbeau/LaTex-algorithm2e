\begin{algorithm}[H]
\caption{CyclicStrategy for Fluct (osc) Head}
\KwInput{Period $T$ (default $6$), bucket capacity $C$, new frame $(t, K_t, V_t)$}
\KwOutput{Anchors for query frame $t_q$ inside $(t_{\mathrm{sink}}, t_{\mathrm{recent}})$}
\KwInit{$T$ FIFO buckets: $\mathrm{bucket}[0],\dots,\mathrm{bucket}[T-1]$}
\SetKwProg{Fn}{Function}{}{}
\Fn{\textup{\textsc{Update}}$(t, K_t, V_t)$}{
  $\phi \gets t \bmod T$\;
  $\mathrm{push\_back}(\mathrm{bucket}[\phi], (t, K_t, V_t))$\;
  \If{$|\mathrm{bucket}[\phi]| > C$}{
    $\mathrm{pop\_front}(\mathrm{bucket}[\phi])$\;
  }
  \Complexity{Update: $\Oh(1)$}
}
\Fn{\textup{\textsc{Collect}}$(t_q, t_{\mathrm{sink}}, t_{\mathrm{recent}})$}{
  $\phi_q \gets t_q \bmod T$\;
  $\mathcal{A} \gets [\ ]$\;
  \ForEach{$(t, K, V) \in \mathrm{bucket}[\phi_q]$}{
    \If{$t_{\mathrm{sink}} < t < t_{\mathrm{recent}}$}{
      $\mathrm{push\_back}(\mathcal{A}, (t, K, V))$\;
    }
  }
  \KwRet $\mathcal{A}$\;
  \Complexity{Collect: worst-case $\Oh(C)$}
}
\end{algorithm}

\begin{algorithm}[H]
\caption{StrideStrategy for Compact (stable) Head}
\KwInput{Stride $K$ (default $6$), capacity $M$ ($-1$ means unlimited), new frame $(t, K_t, V_t)$}
\KwOutput{Anchors inside $(t_{\mathrm{sink}}, t_{\mathrm{recent}})$}
\KwInit{Ordered map $\mathrm{store}: t \mapsto (K_t, V_t)$}
\SetKwProg{Fn}{Function}{}{}
\Fn{\textup{\textsc{Update}}$(t, K_t, V_t)$}{
  \If{$t \bmod K \neq 0$}{
    \KwRet\;
  }
  $\mathrm{store}[t] \gets (K_t, V_t)$\;
  \If{$M \neq -1$ \textbf{and} $|\mathrm{store}| > M$}{
    $t_{\min} \gets \arg\min\limits_{u \in \mathrm{keys}(\mathrm{store})} u$\;
    $\mathrm{delete}(\mathrm{store}[t_{\min}])$\;
  }
  \Complexity{Update: $\Oh(1)$ amortized ($\Oh(\log n)$ with tree map)}
}
\Fn{\textup{\textsc{Collect}}$(t_{\mathrm{sink}}, t_{\mathrm{recent}})$}{
  $\mathcal{A} \gets [\ ]$\;
  \ForEach{$t$ in ascending order of $\mathrm{keys}(\mathrm{store})$}{
    \If{$t_{\mathrm{sink}} < t < t_{\mathrm{recent}}$}{
      $\mathrm{push\_back}(\mathcal{A}, (t, \mathrm{store}[t].K, \mathrm{store}[t].V))$\;
    }
  }
  \KwRet $\mathcal{A}$\;
  \Complexity{Collect: $\Oh(\#\mathrm{selected})$}
}
\end{algorithm}

\hline

\hline
Retrieved anchors & $\leq C$ & $\leq M$ (or $\lfloor \Delta t / K \rfloor$ when unbounded) \\
\hline
Design motivation & Match 6-frame periodic oscillation of attention & Cover long temporal span using sparse stable anchors \\
