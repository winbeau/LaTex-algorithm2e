\begin{algorithm}[H]
\caption{CyclicStrategy \& Attention for Fluct (osc) Head}
\KwInput{Period $T$, bucket capacity $C$, query time $t_q$, new frame $(t, K_t, V_t)$}
\KwData{Sink state $S_{\mathrm{sink}}$, Recent state $S_{\mathrm{recent}}$}
\KwInit{$T$ FIFO buckets: $\mathrm{bucket}[0],\dots,\mathrm{bucket}[T-1]$}
\SetKwProg{Fn}{Function}{}{}

\Fn{\textup{UPDATE}$(t, K_t, V_t)$}{
  $\phi \gets t \bmod T$\;
  $\mathrm{push\_back}(\mathrm{bucket}[\phi], (t, K_t, V_t))$\;
  \If{$|\mathrm{bucket}[\phi]| > C$}{
    $\mathrm{pop\_front}(\mathrm{bucket}[\phi])$\;
  }
  \tcp*[r]{Update: $\mathcal{O}(1)$}
}

\Fn{\textup{COLLECT}$(t_q, t_{\mathrm{sink}}, t_{\mathrm{recent}})$}{
  $\phi_q \gets t_q \bmod T$\;
  $\mathcal{A} \gets [\ ]$\;
  \ForEach{$(t, K, V) \in \mathrm{bucket}[\phi_q]$}{
    \If{$t_{\mathrm{sink}} < t < t_{\mathrm{recent}}$}{
      $\mathrm{push\_back}(\mathcal{A}, (t, K, V))$\;
    }
  }
  \KwRet $\mathcal{A}$\;
  \tcp*[r]{Collect: worst-case $\mathcal{O}(C)$}
}

\Fn{\textup{COMPUTEATTENTION}$(Q_{t_q})$}{
  $\mathcal{A}_{\mathrm{mid}} \gets \text{COLLECT}(t_q, t_{\mathrm{sink}}, t_{\mathrm{recent}})$\;
  $KV_{\mathrm{total}} \gets \mathrm{Concat}(S_{\mathrm{sink}}, \mathcal{A}_{\mathrm{mid}}, S_{\mathrm{recent}})$\;
  $KV_{\mathrm{flat}}, \mathrm{cu\_seqlens} \gets \mathrm{FlattenWithPrefixSum}(KV_{\mathrm{total}})$\;
  \KwRet $\mathrm{VarLenFlashAttn}(Q_{t_q}, KV_{\mathrm{flat}}, \mathrm{cu\_seqlens})$\;
}
\end{algorithm}

\begin{algorithm}[H]
\caption{StrideStrategy \& Attention for Stable (sta+) Head}
\KwInput{Stride length $L$, queue capacity $C$, query time $t_q$, new frame $(t, K_t, V_t)$}
\KwData{Sink state $S_{\mathrm{sink}}$, Recent state $S_{\mathrm{recent}}$}
\KwInit{1 FIFO queue: $\mathrm{queue\_mid}$}
\SetKwProg{Fn}{Function}{}{}

\Fn{\textup{UPDATE}$(t, K_t, V_t)$}{
  \If{$t \geq t_{\mathrm{sink}}$ \textup{\textbf{and}} $(t - t_{\mathrm{sink}}) \bmod L == 0$}{
    $\mathrm{push\_back}(\mathrm{queue\_mid}, (t, K_t, V_t))$\;
    \If{$|\mathrm{queue\_mid}| > C$}{
      $\mathrm{pop\_front}(\mathrm{queue\_mid})$\;
    }
  }
  \tcp*[r]{Update: $\mathcal{O}(1)$}
}

\Fn{\textup{COLLECT}$(t_{\mathrm{sink}}, t_{\mathrm{recent}})$}{
  $\mathcal{A} \gets [\ ]$\;
  \ForEach{$(t, K, V) \in \mathrm{queue\_mid}$}{
    \If{$t_{\mathrm{sink}} < t < t_{\mathrm{recent}}$}{
      $\mathrm{push\_back}(\mathcal{A}, (t, K, V))$\;
    }
  }
  \KwRet $\mathcal{A}$\;
  \tcp*[r]{Collect: worst-case $\mathcal{O}(C)$}
}

\Fn{\textup{COMPUTEATTENTION}$(Q_{t_q})$}{
  $\mathcal{A}_{\mathrm{mid}} \gets \text{COLLECT}(t_{\mathrm{sink}}, t_{\mathrm{recent}})$\;
  $KV_{\mathrm{total}} \gets \mathrm{Concat}(S_{\mathrm{sink}}, \mathcal{A}_{\mathrm{mid}}, S_{\mathrm{recent}})$\;
  $KV_{\mathrm{flat}}, \mathrm{cu\_seqlens} \gets \mathrm{FlattenWithPrefixSum}(KV_{\mathrm{total}})$\;
  \KwRet $\mathrm{VarLenFlashAttn}(Q_{t_q}, KV_{\mathrm{flat}}, \mathrm{cu\_seqlens})$\;
}
\end{algorithm}
